# Phase 3B Inspector Offline UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a separate queue-first React/Vite inspector app with local offline outbox simulation and sync API integration.

**Architecture:** The inspector app lives in `web/apps/inspector` as its own React/Vite surface. It uses feature-local TypeScript domain types, a sync client for Phase 3A endpoints, a `localStorage` outbox adapter, and a mobile-first app shell with Work, Capture, Queue, and Scan-unavailable views. Backend code is not changed in this phase.

**Tech Stack:** React 19, TypeScript, Vite, Vitest, Testing Library, `lucide-react`, FastAPI sync API, browser `localStorage`.

Create `outboxStore.test.ts` with tests for add/update, mark-applied, mark-conflict, and corrupt storage reset:

```ts
import { beforeEach, describe, expect, it } from "vitest";
import {
  createInspectionOperation,
  loadOutbox,
  markOperationApplied,
  markOperationConflict,
  saveOutbox
} from "../offline/outboxStore";

beforeEach(() => {
  localStorage.clear();
});

describe("outboxStore", () => {
  it("adds an inspection operation and persists it", () => {
    const operation = createInspectionOperation({
      assetId: "asset-1",
      inspectionId: "local-inspection-1",
      baseVersion: 2,
      status: "DRAFT",
      appliedPressureKpa: 30200,
      requiredPressureKpa: 20000,
      holdTimeSeconds: 300,
      passed: true,
      notes: "No leaks"
    });

    saveOutbox([operation]);

    expect(loadOutbox().operations).toHaveLength(1);
    expect(loadOutbox().operations[0]).toMatchObject({
      entity: "Inspection",
      entityId: "local-inspection-1",
      status: "pending"
    });
  });

  it("marks operations applied with the server version", () => {
    const operation = createInspectionOperation({
      assetId: "asset-1",
      inspectionId: "local-inspection-1",
      baseVersion: 2,
      status: "SUBMITTED",
      appliedPressureKpa: 30200,
      requiredPressureKpa: 20000,
      holdTimeSeconds: 300,
      passed: true,
      notes: "Submitted"
    });
    saveOutbox([operation]);

    markOperationApplied(operation.opId, 3);

    expect(loadOutbox().operations[0]).toMatchObject({ status: "applied", serverVersion: 3 });
  });

  it("marks conflicts with current server payload", () => {
    const operation = createInspectionOperation({
      assetId: "asset-1",
      inspectionId: "local-inspection-1",
      baseVersion: 2,
      status: "SUBMITTED",
      appliedPressureKpa: 30200,
      requiredPressureKpa: 20000,
      holdTimeSeconds: 300,
      passed: true,
      notes: "Submitted"
    });
    saveOutbox([operation]);

    markOperationConflict(operation.opId, {
      currentVersion: 4,
      payload: { result: "server copy" }
    });

    expect(loadOutbox().operations[0]).toMatchObject({
      status: "conflict",
      currentVersion: 4,
      serverPayload: { result: "server copy" }
    });
  });

  it("resets corrupt stored JSON and reports a warning", () => {
    localStorage.setItem("bat-hms-inspector-outbox", "{bad");

    expect(loadOutbox()).toMatchObject({
      operations: [],
      warning: "Local queue was reset because stored data was unreadable."
    });
  });
});
```

export function createSyncClient(options: { fetchImpl?: typeof fetch } = {}) {
  const fetchImpl = options.fetchImpl ?? fetch;
  return {
    bootstrap: () => requestJson<SyncBootstrapResponse>(fetchImpl, "/api/v1/sync/bootstrap"),
    changes: (since: number) => requestJson<SyncChangesResponse>(fetchImpl, `/api/v1/sync/changes?since=${since}`),
    push: (operations: OutboxOperation[]) => requestJson<SyncPushResponse>(fetchImpl, "/api/v1/sync/push", {
      method: "POST",
      body: JSON.stringify({ operations: operations.map(toApiOperation) })
    })
  };
}
